# Employee Performance Analysis

import pandas as pd
import matplotlib.pyplot as plt
import mpld3

# Sample dataset
data = {
    "EmployeeID": range(1, 101),
    "Department": ["Operations", "HR", "Finance", "IT", "Marketing"]*20,
    "Region": ["North", "South", "East", "West", "Central"]*20
}

df = pd.DataFrame(data)

# Frequency count for 'Operations' department
ops_count = (df["Department"] == "Operations").sum()
print("Frequency Count for 'Operations' Department:", ops_count)

# Histogram of department distribution
plt.figure(figsize=(8,5))
plt.hist(df["Department"], bins=len(df["Department"].unique()), color='skyblue', edgecolor='black')
plt.title("Department Distribution Histogram")
plt.xlabel("Department")
plt.ylabel("Count")
plt.text(0.5, -0.25, "Analysis by: 23f2003561@ds.study.iitm.ac.in",
         ha='center', va='top', transform=plt.gca().transAxes)

# Save interactive plot as HTML
mpld3.save_html(plt.gcf(), "employee_analysis.html")

import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import mpld3

data = pd.read_csv("employee_data.csv")
data['department'] = data['department'].str.strip().str.title()
operations_count = data[data['department'] == 'Operations'].shape[0]
print(f"Number of employees in Operations department: {operations_count}")

plt.figure(figsize=(10,6))
sns.countplot(data=data, x='department', palette='viridis')
plt.title('Employee Distribution Across Departments\n(Analysis by 23f2003561@ds.study.iitm.ac.in)')
plt.xlabel('Department')
plt.ylabel('Number of Employees')
plt.xticks(rotation=45)
fig = plt.gcf()
